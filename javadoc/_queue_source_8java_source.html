142  @Override
143  @SuppressWarnings("squid:S1168")
144  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
145  {
146  int size = m_events.size();
147  if (m_index >= size)
148  {
149  return false;
150  }
151  Object[] output = new Object[getOutputArity()];
152  Object event = m_events.get(m_index);
153  if (m_loop)
154  {
155  m_index = (m_index + 1) % size;
156  }
157  else
158  {
159  // If we don't loop, play the events only once
160  m_index++;
161  }
162  if (m_index > size && !m_loop)
163  {
164  // No more events from this queue
165  return false;
166  }
167  for (int i = 0; i < getOutputArity(); i++)
168  {
169  if (event == null)
170  {
171  // If one of the elements is null, don't output anything
172  return true;
173  }
174  output[i] = event;
175  }
176  outputs.add(output);
177  if (m_eventTracker != null)
178  {
179  int index = m_index - 1;
180  if (index < 0)
181  {
182  index += size;
183  }
184  for (int i = 0; i < getOutputArity(); i++)
185  {
186  associateTo(new QueueFunction(getId(), index), i, m_outputCount);
187  }
188  }
189  m_outputCount++;
190  return true;
191  }
