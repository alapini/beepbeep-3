140  @Override
141  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
142  {
143  int output_arity = getOutputArity();
144  Object[] f_value = new Object[1];
145  try
146  {
147  m_slicingFunction.evaluate(inputs, f_value);
148  }
149  catch (FunctionException e)
150  {
151  throw new ProcessorException(e);
152  }
153  Object slice_ids = f_value[0];
154  if (slice_ids == null)
155  {
156  // This event applies to no slice; don't bother processing it
157  return produceReturn(outputs);
158  }
159  Object[] slice_vals;
160  if (m_explodeArrays)
161  {
162  if (slice_ids.getClass().isArray())
163  {
164  slice_vals = (Object[]) slice_ids;
165  }
166  if (slice_ids instanceof Collection)
167  {
168  Collection<?> col = (Collection<?>) slice_ids;
169  slice_vals = new Object[col.size()];
170  int i = 0;
171  for (Object o : col)
172  {
173  slice_vals[i] = o;
174  i++;
175  }
176  }
177  else
178  {
179  slice_vals = new Object[] { slice_ids };
180  }
181  }
182  else
183  {
184  slice_vals = new Object[] { slice_ids };
185  }
186  for (Object slice_id : slice_vals)
187  {
188  Set<Object> slices_to_process = new HashSet<Object>();
189  if (slice_id instanceof ToAllSlices || slice_id == null)
190  {
191  slices_to_process.addAll(m_slices.keySet());
192  }
193  else
194  {
195  if (!m_slices.containsKey(slice_id))
196  {
197  // First time we see this value: create new slice
198  Processor p = m_processor.duplicate();
199  m_slices.put(slice_id, p);
200  addContextFromSlice(p, slice_id);
201  QueueSink sink = new QueueSink(output_arity);
202  Connector.connect(p, sink);
203  m_sinks.put(slice_id, sink);
204  }
205  slices_to_process.add(slice_id);
206  }
207  @SuppressWarnings("unchecked")
208  Future<Pushable>[] futures = new Future[slices_to_process.size()];
209  for (Object s_id : slices_to_process)
210  {
211  // Find processor corresponding to that slice
212  Processor slice_p = m_slices.get(s_id);
213  // If this slice hasn't been cleaned up...
214  if (slice_p != null)
215  {
216  QueueSink sink_p = m_sinks.get(s_id);
217  // Push the input into the processor
218  // Pushable[] p_array = new Pushable[inputs.length];
219  for (int i = 0; i < inputs.length; i++)
220  {
221  Object o_i = inputs[i];
222  Pushable p = slice_p.getPushableInput(i);
223  Future<Pushable> f = p.pushFast(o_i);
224  futures[i] = f;
225  // p_array[i] = p;
226  }
227  // Wait for all slices to be done
228  for (Future<Pushable> f : futures)
229  {
230  if (f != null)
231  {
232  try
233  {
234  f.get();
235  }
236  catch (InterruptedException e)
237  {
238  throw new ProcessorException(e);
239  }
240  catch (ExecutionException e)
241  {
242  throw new ProcessorException(e);
243  }
244  }
245  }
246  // Collect the output from that processor
247  Object[] out = sink_p.remove();
248  // Can we clean that slice?
249  Object[] can_clean = new Object[1];
250  if (m_cleaningFunction != null)
251  {
252  try
253  {
254  m_cleaningFunction.evaluate(out, can_clean);
255  }
256  catch (FunctionException e)
257  {
258  throw new ProcessorException(e);
259  }
260  }
261  if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
262  && (Boolean) (can_clean[0]))
263  {
264  // Yes: remove the processor for that slice
265  m_slices.remove(s_id);
266  m_sinks.remove(s_id);
267  }
268  handleNewSliceValue(s_id, out[0], outputs);
269  }
270  }
271  }
272  return produceReturn(outputs);
273  }
