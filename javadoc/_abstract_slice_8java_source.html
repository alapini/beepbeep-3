149  @Override
150  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
151  {
152  int output_arity = getOutputArity();
153  Object[] f_value = new Object[1];
154  try
155  {
156  m_slicingFunction.evaluate(inputs, f_value);
157  }
158  catch (FunctionException e)
159  {
160  throw new ProcessorException(e);
161  }
162  Object slice_ids = f_value[0];
163  if (slice_ids == null)
164  {
165  // This event applies to no slice; don't bother processing it
166  return produceReturn(outputs);
167  }
168  Object[] slice_vals;
169  if (m_explodeArrays)
170  {
171  if (slice_ids.getClass().isArray())
172  {
173  slice_vals = (Object[]) slice_ids;
174  }
175  if (slice_ids instanceof Collection)
176  {
177  Collection<?> col = (Collection<?>) slice_ids;
178  slice_vals = new Object[col.size()];
179  int i = 0;
180  for (Object o : col)
181  {
182  slice_vals[i] = o;
183  i++;
184  }
185  }
186  else
187  {
188  slice_vals = new Object[] { slice_ids };
189  }
190  }
191  else
192  {
193  slice_vals = new Object[] { slice_ids };
194  }
195  for (Object slice_id : slice_vals)
196  {
197  Set<Object> slices_to_process = new HashSet<Object>();
198  if (slice_id instanceof ToAllSlices || slice_id == null)
199  {
200  slices_to_process.addAll(m_slices.keySet());
201  }
202  else
203  {
204  if (!m_slices.containsKey(slice_id))
205  {
206  // First time we see this value: create new slice
207  Processor p = m_processor.duplicate();
208  m_slices.put(slice_id, p);
209  addContextFromSlice(p, slice_id);
210  QueueSink sink = new QueueSink(output_arity);
211  Connector.connect(p, sink);
212  m_sinks.put(slice_id, sink);
213  if (m_eventTracker != null)
214  {
215  p.setEventTracker(m_eventTracker.getCopy());
216  m_sliceIndices.put(slice_id, new ArrayList<Integer>());
217  }
218  }
219  slices_to_process.add(slice_id);
220  }
221  @SuppressWarnings("unchecked")
222  Future<Pushable>[] futures = new Future[slices_to_process.size()];
223  for (Object s_id : slices_to_process)
224  {
225  // Find processor corresponding to that slice
226  Processor slice_p = m_slices.get(s_id);
227  // If this slice hasn't been cleaned up...
228  if (slice_p != null)
229  {
230  QueueSink sink_p = m_sinks.get(s_id);
231  if (m_eventTracker != null)
232  {
233  m_sliceIndices.get(s_id).add(m_inputCount);
234  }
235  // Push the input into the processor
236  // Pushable[] p_array = new Pushable[inputs.length];
237  for (int i = 0; i < inputs.length; i++)
238  {
239  Object o_i = inputs[i];
240  Pushable p = slice_p.getPushableInput(i);
241  Future<Pushable> f = p.pushFast(o_i);
242  futures[i] = f;
243  // p_array[i] = p;
244  }
245  // Wait for all slices to be done
246  for (Future<Pushable> f : futures)
247  {
248  if (f != null)
249  {
250  try
251  {
252  f.get();
253  }
254  catch (InterruptedException e)
255  {
256  throw new ProcessorException(e);
257  }
258  catch (ExecutionException e)
259  {
260  throw new ProcessorException(e);
261  }
262  }
263  }
264  // Collect the output from that processor
265  Object[] out = sink_p.remove();
266  // Can we clean that slice?
267  Object[] can_clean = new Object[1];
268  if (m_cleaningFunction != null)
269  {
270  try
271  {
272  m_cleaningFunction.evaluate(out, can_clean);
273  }
274  catch (FunctionException e)
275  {
276  throw new ProcessorException(e);
277  }
278  }
279  if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
280  && (Boolean) (can_clean[0]))
281  {
282  // Yes: remove the processor for that slice
283  m_slices.remove(s_id);
284  m_sinks.remove(s_id);
285  }
286  handleNewSliceValue(s_id, out[0], outputs);
287  }
288  }
289  }
290  m_inputCount++;
291  return produceReturn(outputs);
292  }
