483  {
484  if (x instanceof Number)
485  {
486  return (Number) x;
487  }
488  if (!(x instanceof String))
489  {
490  // Anything but a string: work on the value of toString
491  x = x.toString();
492  }
493  if (x instanceof String)
494  {
495  try
496  {
497  return Integer.parseInt((String) x);
498  }
499  catch (NumberFormatException e)

501  try

503  return Float.parseFloat((String) x);
504  }
505  catch (NumberFormatException e2)

509  }
510  }
