804  @SuppressWarnings("unchecked")
805  @Override
806  public final Processor read(ObjectReader<?> reader, Object o) throws ProcessorException
807  {
808  Map<String, Object> contents = null;
809  try
810  {
811  contents = (Map<String,Object>) reader.read(o);
812  }
813  catch (ReadException e)
814  {
815  throw new ProcessorException(e);
816  }
817  Processor p = null;
818  if (contents.containsKey("contents"))
819  {
820  Object o_contents = contents.get("contents");
821  try
822  {
823  p = readState(o_contents);
824  }
825  catch (UnsupportedOperationException e)
826  {
827  throw new ProcessorException(e);
828  }
829  }
830  if (p == null)
831  {
832  throw new ProcessorException("The processor returned null with being deserialized");
833  }
834  p.m_inputCount = ((Number) contents.get("input-count")).intValue();
835  p.m_outputCount = ((Number) contents.get("output-count")).intValue();
836  try
837  {
838  ObjectReader.setField(p, "m_uniqueId", ((Number) contents.get("id")).intValue());
839  }
840  catch (ReadException e)
841  {
842  throw new ProcessorException(e);
843  }
844  List<Queue<Object>> in_queues = (List<Queue<Object>>) contents.get("input-queues");
845  for (int i = 0; i < in_queues.size(); i++)
846  {
847  p.m_inputQueues[i] = in_queues.get(i);
848  }
849  List<Queue<Object>> out_queues = (List<Queue<Object>>) contents.get("output-queues");
850  for (int i = 0; i < out_queues.size(); i++)
851  {
852  p.m_outputQueues[i] = in_queues.get(i);
853  }
854  return p;
855  }
